85/* Function: DType2Real_T ======================================================



86 * Convert a built-in data type to a real_T value.  Return the real_T value



87 * as well as a string (by reference) corresponding to the original data type.



88 * If the data type is not recognized as a Simulink built-in data type, then



89 * NULL is returned for the dTypeName and 0 is returned for the dTypeValue.



90 */



91PRIVATE double DType2Double(



92    const char              *vPtr,



93    const int               dTypeIdx,



94    const DataTypeTransInfo *dtInfo,



95    const char              **dTypeName)



96{



97    real_T     outVal;



98    char * const *dTypeNames = (char * const *)dtGetDataTypeNames(dtInfo);



99    const char *thisName     = dTypeNames[dTypeIdx];



100



101    *dTypeName = thisName;



102



103    if (strcmp(thisName, "real_T") == 0) {



104        outVal = (real_T) (((const real_T*)vPtr)[0]);



105    } else if (strcmp(thisName, "real32_T") == 0) {



106        outVal = (real_T) (((const real32_T*)vPtr)[0]);



107    } else if (strcmp(thisName, "int8_T") == 0) {



108        outVal = (real_T) (((const int8_T*)vPtr)[0]);



109    } else if (strcmp(thisName, "uint8_T") == 0) {



110        outVal = (real_T) (((const uint8_T*)vPtr)[0]);



111    } else if (strcmp(thisName, "int16_T") == 0) {



112        outVal = (real_T) (((const int16_T*)vPtr)[0]);



113    } else if (strcmp(thisName, "uint16_T") == 0) {



114        outVal = (real_T) (((const uint16_T*)vPtr)[0]);



115    } else if (strcmp(thisName, "int32_T") == 0) {



116        outVal = (real_T) (((const int32_T*)vPtr)[0]);



117    } else if (strcmp(thisName, "uint32_T") == 0) {



118        outVal = (real_T) (((const uint32_T*)vPtr)[0]);



119    } else if (strcmp(thisName, "boolean_T") == 0) {



120        outVal = (real_T) (((const boolean_T*)vPtr)[0]);



121    } else {



122        outVal    = 0;



123        dTypeName = NULL;



124    }



125    return(outVal);



126} /* end DType2Double */

130/* Function: SetParam ==========================================================



131 * Install new parameters.



132 *



133 * NOTE: pbuf looks like:



134 *  [NPARAMS



135 *   B S W DI DATA   % pVal 0



136 *   B S W DI DATA   % pVal 1



137 *   B S W DI DATA   % pVal N



138 *  ]



139 *



140 *  No assumptions about the alignment of pbuf may be made.



141 *



142 *  where,



143 *      B   : Data type transition index.  Note that on the target the data



144 *            type transition index provides both the base address (B)



145 *              of the transition as well as whether or not



146 *              that section of the array contains complex elements.



147 *      S   : starting offset of the data from the DT_TRANS_IDX



148 *      W   : number of elements for this param



149 *      DI  : index into rtw data type table (inUse idx)



150 *      DATA: the param values (in target format).



151 *



152 *  and,



153 *      All values, excluding DATA, are int32_T.



154 */



155#ifndef EXTMODE_DISABLEPARAMETERTUNING



156PUBLIC void SetParam(RTWExtModeInfo  *ei, const char *pbuf)



157{



158    int        i;



159    int32_T    nParams;



160    const char *bufPtr    = pbuf;



161    const int  B          = 0; /* index into dtype tran table (base address)  */



162    const int  SI         = 1; /* starting index - wrt to base address        */



163    const int  W          = 2; /* width of section (number of elements)       */



164    const int  DI         = 3; /* index into data type tables                 */



165    const int  tmpBufSize = sizeof(int32_T) * 4;



166    int32_T    tmpBuf[4];



167



168    const DataTypeTransInfo *dtInfo = (const DataTypeTransInfo *) rteiGetModelMappingInfo(ei);



169    const DataTypeTransitionTable *dtTable = dtGetParamDataTypeTrans(dtInfo);



170    const uint_T *dtSizes = dtGetDataTypeSizes(dtInfo);



171



172    /* unpack NPARAMS */



173    (void)memcpy(&nParams, bufPtr, sizeof(int32_T));



174    bufPtr += sizeof(int32_T);



175    



176#ifdef VERBOSE



177    printf("\nUpdating %d parameters....\n", nParams);



178#endif



179



180    /*



181     * Unpack the data and install the new parameters.



182     */



183    for (i=0; i<nParams; i++) {



184        int_T   elSize;



185        int_T   nBytes;



186        char_T  *start;



187        char_T  *tranAddress;



188        int_T   tranIsComplex;



189



190        /* unpack B SI W DI */



191        (void)memcpy(tmpBuf, bufPtr, tmpBufSize);



192        bufPtr += tmpBufSize;



193



194        /* 



195         * Find starting address and size (nBytes) for this parameters 



196         * section of memory.



197         */



198        tranAddress   = dtTransGetAddress(dtTable, tmpBuf[B]);



199        tranIsComplex = dtTransGetComplexFlag(dtTable, tmpBuf[B]);



200



201        elSize = dtSizes[tmpBuf[DI]] * (tranIsComplex ? 2 : 1);



202        nBytes = tmpBuf[W] * elSize;



203        start = tranAddress + (tmpBuf[SI] * elSize);



204



205        /* Install the params. */



206        (void)memcpy(start, bufPtr, nBytes);



207        bufPtr += nBytes;



208



209#ifdef VERBOSE



210        /*



211         * It is safe to assume that once the params are installed into



212         * the param vector that they are properly aligned.  So we



213         * do our verbosity print-out here.



214         */



215        {



216            double     val;



217            const char *dTypeName;



218            const char *vPtr = (const char *)start;



219



220            val = DType2Double(start, tmpBuf[DI], dtInfo, &dTypeName);



221            printf("\n\tParam| "



222                   "DT_Trans: %d, index: %d, nEls: %d, data type: [%s, %d]\n",



223                   tmpBuf[B], tmpBuf[SI], tmpBuf[W],



224                   (dTypeName != NULL) ? dTypeName : "",tmpBuf[DI]);



225



226            if (!tranIsComplex) {



227                int j;



228                for (j=0; j<tmpBuf[W]; j++) {



229                    val = DType2Double(vPtr, tmpBuf[DI], dtInfo, &dTypeName);



230                    printf("\t\t%f\n", val);



231                    vPtr += dtSizes[tmpBuf[DI]];



232                }



233            } else {



234                int j;



235                for (j=0; j<tmpBuf[W]; j++) {



236                    val = DType2Double(vPtr, tmpBuf[DI], dtInfo, &dTypeName);



237                    printf("\t\t%f + ", val);



238                    vPtr += dtSizes[tmpBuf[DI]];



239



240                    val = DType2Double(vPtr, tmpBuf[DI], dtInfo, &dTypeName);



241                    printf("\t\t%fi\n", val);



242                    vPtr += dtSizes[tmpBuf[DI]];



243                }



244            }



245        }



246#endif



247    }



248} /* end SetParam */



249#endif /* ifndef EXTMODE_DISABLEPARAMETERTUNING */

1428/* Function ====================================================================



1429 * Assign sections in the circular buffer for the requested number of bytes



1430 * (i.e., fill in the bufMem struct).  If there is no room in the circular



1431 * buffer return an overflow error.



1432 *



1433 * NOTE: Do not move the CircularBuffers head forward in this function!  



1434 *       Only move the tmpHead forward.  The actual head is not advanced



1435 *       until the entire time point is successfully copied into the buffer.



1436 *



1437 *       This function modifies tmpHead to point at the next available 



1438 *       location.



1439 *



1440 *       It is possible for tmpHead to equal the tail upon entry to this



1441 *       function.  This does not necessarily mean that the buffer is



1442 *       empty (unwrapped).  It could also mean that the buffer is exactly



1443 *       full (this is considered as wrapped).



1444 */



1445#ifndef EXTMODE_DISABLESIGNALMONITORING



1446PRIVATE boolean_T UploadBufAssignMem(



1447    CircularBuf  *circBuf,



1448    int_T        nBytesToAdd,



1449    char         **tmpHead,   /* in-out */



1450    BufMem       *bufMem)     /* out */



1451{



1452    int_T       nBytesLeft;



1453    boolean_T   overFlow  = false;



1454    char        *end      = circBuf->buf + circBuf->bufSize; /* 1 passed end */



1455



1456    if ((*tmpHead > circBuf->tail) || circBuf->empty) {



1457        /* buffer not wrapped */



1458        nBytesLeft = (int_T)((end - *tmpHead) + (circBuf->tail - circBuf->buf));



1459



1460        if (nBytesLeft < nBytesToAdd) {



1461            overFlow = true;



1462            goto EXIT_POINT;



1463        }



1464



1465        if ((*tmpHead + nBytesToAdd) < end) {



1466            /* still not wrapped */



1467            bufMem->nBytes1  = nBytesToAdd;



1468            bufMem->section1 = *tmpHead;



1469



1470            bufMem->nBytes2  = 0;



1471            bufMem->section2 = NULL;



1472



1473            *tmpHead += nBytesToAdd;



1474        } else {



1475            /* now we're wrapped */



1476            bufMem->nBytes1  = (int_T)(end - *tmpHead);



1477            bufMem->section1 = *tmpHead;



1478



1479            bufMem->nBytes2  = nBytesToAdd - bufMem->nBytes1;



1480            bufMem->section2 = (bufMem->nBytes2 > 0) ? circBuf->buf : NULL;



1481



1482            *tmpHead = circBuf->buf + bufMem->nBytes2;



1483        }  



1484    } else {



1485        /* wrapped */



1486        nBytesLeft = (int_T)(circBuf->tail - *tmpHead);



1487        if (nBytesLeft < nBytesToAdd) {



1488            overFlow = true;



1489            goto EXIT_POINT;



1490        }



1491



1492        bufMem->nBytes1  = nBytesToAdd;



1493        bufMem->section1 = *tmpHead;



1494



1495        bufMem->nBytes2  = 0;



1496        bufMem->section2 = NULL;



1497



1498        *tmpHead += nBytesToAdd;



1499    }



1500    



1501EXIT_POINT:



1502    return(overFlow);



1503} /* end UploadBufAssignMem */



1504#endif /* ifndef EXTMODE_DISABLESIGNALMONITORING */

1559/* Function ====================================================================



1560 * If the trigger is in the TRIGGER_FIRED state or we are collecting data for



1561 * pre-triggering, add data, for each tid with a hit, to the upload buffers.  



1562 * This function is called from within the appropriate task, once per sample



1563 * hit.



1564 *



1565 * The format of the packet that is sent to the host is as follows:



1566 *



1567 * definitions:



1568 *      pktType - A qualifier indicating any special action that needs to be



1569 *                taken (e.g., a termination flag following the last data point,



1570 *                or a flag indicating that it is the first data point after



1571 *                a trigger event).



1572 *



1573 *      nBytes - total number of target bytes for this packet (including the



1574 *               nBytes field).  nBytes worth of data represents 1 full time



1575 *               step of the target simulation.



1576 *



1577 *      nSys - The number of systems for which this packet contains data.



1578 *



1579 *      tid - The tid with which this data is associated.



1580 *



1581 *      upInfoIdx - upInfo index



1582 *



1583 *      t - simulation time



1584 *



1585 *      sysId - The index into the BdUploadInfo.sysTables array so that we can



1586 *              map the target data back to the appropriate system.  This is



1587 *              NOT the descendant system index!



1588 *



1589 *      data - the target simulation data (in target format)



1590 *



1591 * The packet looks like:



1592 * [nBytes pktType nSys tid upInfoIdx t sysId [data] sysId [data]...]



1593 *     |                            | |         | |          |



1594 *     ----------------------------- ----------- ------------



1595 *          pkt header          sys data     sys data



1596 *



1597 * Ints are int32_T.



1598 */



1599#ifndef EXTMODE_DISABLESIGNALMONITORING



1600PUBLIC void UploadBufAddTimePoint(int_T tid, real_T taskTime,



1601                                  int32_T upInfoIdx)



1602{



1603    int_T        preTrig;



1604    int_T        overFlow;



1605    TriggerInfo  *trigInfo;



1606    CircularBuf  *circBuf;



1607    BdUploadInfo *uploadInfo = &uploadInfoArray[upInfoIdx];



1608



1609    overFlow   = false;



1610    trigInfo   = &uploadInfo->trigInfo;



1611    circBuf    = &uploadInfo->circBufs[tid];



1612    



1613    /*



1614     * Check for transitions from the TRIGGER_ARMED state to either the



1615     * TRIGGER_FIRED_STATE or the TRIGGER_DELAYED state.  We only do this



1616     * if it is a sample hit for the trigger signal.  Note that this



1617     * is the only place in the whole world that the trigger state can



1618     * move from TRIGGER_ARMED_STATE to TRIGGER_DELAYED or TRIGGER_FIRED.



1619     */



1620    if (trigInfo->state == TRIGGER_ARMED) {



1621        if (trigInfo->trigSignals.nSections == 0) {



1622            /* short-circuit for manual trigger */



1623            trigInfo->state = TRIGGER_FIRED;



1624        } else



1625            if ((tid == trigInfo->tid) &&



1626                (UploadCheckTriggerSignals(upInfoIdx))) {



1627                /* trig signal crossing */



1628                if (trigInfo->delay == 0) {



1629                    trigInfo->state = TRIGGER_FIRED;



1630                    /* 0 unless pre-trig */



1631                    trigInfo->count = trigInfo->preTrig.count;



1632                } else {



1633                    trigInfo->state = TRIGGER_DELAYED;



1634                    assert(trigInfo->count == 0);



1635                    



1636                    /* We will be skipping this step, so the delay count is 1. */



1637                    trigInfo->count = 1;



1638                }



1639            }



1640    }



1641    



1642    preTrig = (trigInfo->state == TRIGGER_ARMED) &&



1643        (trigInfo->preTrig.duration > 0);



1644    



1645    /*



1646     * Handle adding data to the collection buffers - if needed.



1647     */



1648    if (((trigInfo->state == TRIGGER_FIRED) || preTrig) &&



1649        /* bufSize == 0 means no signals in this tid */



1650        circBuf->bufSize != 0) {



1651        



1652        int32_T     i;



1653        BufMem      bufMem;



1654        BufMem      pktStart;



1655        int_T       size;



1656        char_T *tmpHead    = circBuf->head;



1657        const int_T PKT_TYPE_IDX = 0;



1658        const int_T NBYTES_IDX   = 1;



1659        const int_T NSYS_IDX     = 2;



1660        const int_T TID_IDX      = 3;



1661        const int_T UPINFO_IDX   = 4;



1662



1663        int32_T intHdr[5] = {0, 0, 0, 0, 0};



1664        intHdr[UPINFO_IDX] = upInfoIdx;



1665        



1666        if (preTrig && (trigInfo->preTrig.count==trigInfo->preTrig.duration)) {



1667            /* Advance the tail (we don't need the oldest point anymore). */



1668            char *end = circBuf->buf + circBuf->bufSize;



1669            MOVE_TAIL_ONESTEP(circBuf, end);



1670            trigInfo->preTrig.count--;



1671        }



1672        



1673        /*



1674         * Save some space for the 5 integer values that make up the packet



1675         * header: [pktType nBytes nSys tid upInfoIdx].



1676         * The values are filled in later.



1677         */



1678        size = 5*sizeof(int32_T);



1679        overFlow = UploadBufAssignMem(circBuf, size, &tmpHead, &pktStart);



1680        if (overFlow) goto EXIT_POINT;



1681



1682        /*



1683         * We do not want to include the packet type and number of bytes



1684         * in the size calculation.  Size should represent the payload of



1685         * this packet.  The packet type and number of bytes represent the



1686         * packet header and are not included in the payload size.



1687         */



1688        size -= 2*sizeof(int32_T);



1689        intHdr[NBYTES_IDX] += size;



1690        



1691        /* time */



1692        overFlow =



1693            UploadBufAssignMem(circBuf, sizeof(real_T), &tmpHead, &bufMem);



1694        if (overFlow) goto EXIT_POINT;



1695        intHdr[NBYTES_IDX] += sizeof(real_T);



1696        



1697        CIRCBUF_COPY_DATA(bufMem, &taskTime);



1698



1699        /*



1700         * Check each system for an UploadMap. 



1701         */



1702        for (i=0; i<uploadInfo->nSys; i++) {



1703            const SysUploadTable *sysTable =



1704                (const SysUploadTable *)&uploadInfo->sysTables[i];



1705            



1706            if ( (*sysTable->enableState != SUBSYS_RAN_BC_DISABLE) && 



1707                 (*sysTable->enableState != SUBSYS_RAN_BC_ENABLE_TO_DISABLE) ) {



1708                UploadMap *map = sysTable->uploadMap[tid];



1709



1710                if (map != NULL) {



1711                    int_T section;



1712                    intHdr[NSYS_IDX]++;



1713                    



1714                    /* Add system index */



1715                    size = sizeof(int32_T);



1716                    overFlow =



1717                        UploadBufAssignMem(circBuf, size, &tmpHead, &bufMem);



1718                    if (overFlow) goto EXIT_POINT;



1719                    intHdr[NBYTES_IDX] += size;



1720                    



1721                    CIRCBUF_COPY_DATA(bufMem, &i);



1722                    



1723                    /* Add data values */



1724                    for (section=0; section<map->nSections; section++) {



1725                        UploadSection *sect = &map->sections[section];



1726



1727                        overFlow = UploadBufAssignMem(



1728                            circBuf, sect->nBytes, &tmpHead, &bufMem);



1729                        if (overFlow) goto EXIT_POINT;



1730                        intHdr[NBYTES_IDX] += sect->nBytes;



1731                        



1732                        CIRCBUF_COPY_DATA(bufMem, sect->start);



1733                    }



1734                }



1735            }



1736        }



1737



1738        /* If no systems were active then, do nothing. */



1739        if (intHdr[NSYS_IDX] == 0) goto EXIT_POINT;



1740        



1741        /*



1742         * Go back and finish the header: [nBytes pktType nSys tid]



1743         */



1744        



1745        /* ...pktType */



1746        intHdr[PKT_TYPE_IDX] = EXT_UPLOAD_LOGGING_DATA;



1747



1748        /* ...tid */



1749        intHdr[TID_IDX] = tid;



1750        CIRCBUF_COPY_DATA(pktStart, intHdr);



1751



1752        /*



1753         * Time point successfully added to queue.



1754         */



1755        circBuf->head  = tmpHead;



1756        circBuf->empty = false;



1757        



1758        if (preTrig) {



1759            trigInfo->preTrig.count++;



1760        }



1761    }



1762



1763EXIT_POINT:



1764    if (!preTrig) {



1765        if (overFlow) {



1766            trigInfo->overFlow = true;



1767            trigInfo->state    = TRIGGER_TERMINATING;



1768        }



1769#ifdef VXWORKS



1770        else if (trigInfo->state == TRIGGER_FIRED) {



1771            /* allow upload server to run - if data needs to be uploaded */



1772            semGive(uploadSem);



1773        }



1774#endif



1775    } 



1776} /* end UploadBufAddTimePoint */
