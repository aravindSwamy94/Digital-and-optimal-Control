9extern "C" void MPU6050Accel_Init(void)



10{



11    Wire.begin();



12    accelgyro.initialize();



13	//  Enable I2C bypass on MPU6050 so the compass can be accessed



14	I2Cdev::writeBit(MPU6050_DEFAULT_ADDRESS, MPU6050_RA_USER_CTRL, MPU6050_USERCTRL_I2C_MST_EN_BIT, false);



15	I2Cdev::writeBit(MPU6050_DEFAULT_ADDRESS, MPU6050_RA_INT_PIN_CFG, MPU6050_INTCFG_I2C_BYPASS_EN_BIT, true);



16	I2Cdev::writeBit(MPU6050_DEFAULT_ADDRESS, MPU6050_RA_PWR_MGMT_1, MPU6050_PWR1_SLEEP_BIT, false);



17}



18extern "C" void MPU6050Accel_Read(int* pfData)



19{



20    pfData[0]=accelgyro.getAccelerationX();



21    pfData[1]=accelgyro.getAccelerationY();



22    pfData[2]=accelgyro.getAccelerationZ();



23}
