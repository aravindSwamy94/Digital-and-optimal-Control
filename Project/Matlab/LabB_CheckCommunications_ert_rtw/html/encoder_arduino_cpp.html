43// Auxiliary debouncing function



44static void debounce(int del)



45{



46    for (int k = 0; k < del; k++) {



47        // can't use delay in the ISR so need to waste some time



48        // perfoming operations, this uses roughly 0.1ms on uno



49        k = k +0.0 +0.0 -0.0 +3.0 -3.0;



50    }



51}

373// Initialization function called by Encoder System object



374extern "C" void enc_init(int enc, int pinA, int pinB)



375{



376    // enc is the encoder number and it can be 0,1 or 2



377    // if other encoder blocks are present in the model



378    // up to a maximum of 3, they need to refer to a



379    // different encoder number



380    



381    // store pinA and pinB in global encoder structure Enc



382    // they will be needed later by the interrupt routine



383    // that will not be able to access s-function parameters



384    



385    Enc[enc].pinA=pinA;      // set pin A



386    Enc[enc].pinB=pinB;      // set pin B



387    



388    // set encoder pins as inputs



389    pinMode(Enc[enc].pinA, INPUT);



390    pinMode(Enc[enc].pinB, INPUT);



391    



392    // turn on pullup resistors



393    digitalWrite(Enc[enc].pinA, HIGH);



394    digitalWrite(Enc[enc].pinB, HIGH);



395    



396    // attach interrupts



397    switch(enc) {



398        case 0:



399            attachInterrupt(getIntNum(Enc[0].pinA), irsPinAEn0, CHANGE);



400            attachInterrupt(getIntNum(Enc[0].pinB), isrPinBEn0, CHANGE);



401            break;



402        case 1:



403            attachInterrupt(getIntNum(Enc[1].pinA), irsPinAEn1, CHANGE);



404            attachInterrupt(getIntNum(Enc[1].pinB), isrPinBEn1, CHANGE);



405            break;



406        case 2:



407            attachInterrupt(getIntNum(Enc[2].pinA), irsPinAEn2, CHANGE);



408            attachInterrupt(getIntNum(Enc[2].pinB), isrPinBEn2, CHANGE);



409            break;



410		case 3:



411			// attach interrupts for enc[3]:



412			// make analog pin A8 and RX3 (D13) into pin change interrupts:



413			PCICR |= (1 << PCIE2);     // enable PCINT 2



414			PCICR |= (1 << PCIE1);     // enable PCINT 1 



415			PCMSK2 |= (1 << PCINT16);  // Analog 8 as input



416			PCMSK1 |= (1 << PCINT9);  // RX3 D15 as input 



417			break;



418		case 4:



419			// attach interrupts for enc[4]:



420			// ping 11 and 12 into pin change interrupts:



421			PCICR |= (1 << PCIE0);     // enable PCINT0  (pins 11 and 12 --> PCINT5, PCINT6 --> PCINT0 vec)



422			PCMSK0 |= (1 << PCINT5);  // Pin 11



423			PCMSK0 |= (1 << PCINT6);  // Pin 12



424			break;



425			



426    }



427}
