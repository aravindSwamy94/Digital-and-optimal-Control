96/** Read a single bit from an 8-bit device register.



97 * @param devAddr I2C slave device address



98 * @param regAddr Register regAddr to read from



99 * @param bitNum Bit position to read (0-7)



100 * @param data Container for single bit value



101 * @param timeout Optional read timeout in milliseconds (0 to disable, leave off to use default class value in I2Cdev::readTimeout)



102 * @return Status of read operation (true = success)



103 */



104int8_t I2Cdev::readBit(uint8_t devAddr, uint8_t regAddr, uint8_t bitNum, uint8_t *data, uint16_t timeout) {



105    uint8_t b;



106    uint8_t count = readByte(devAddr, regAddr, &b, timeout);



107    *data = b & (1 << bitNum);



108    return count;



109}

126/** Read multiple bits from an 8-bit device register.



127 * @param devAddr I2C slave device address



128 * @param regAddr Register regAddr to read from



129 * @param bitStart First bit position to read (0-7)



130 * @param length Number of bits to read (not more than 8)



131 * @param data Container for right-aligned value (i.e. '101' read from any bitStart position will equal 0x05)



132 * @param timeout Optional read timeout in milliseconds (0 to disable, leave off to use default class value in I2Cdev::readTimeout)



133 * @return Status of read operation (true = success)



134 */



135int8_t I2Cdev::readBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t *data, uint16_t timeout) {



136    // 01101001 read byte



137    // 76543210 bit numbers



138    //    xxx   args: bitStart=4, length=3



139    //    010   masked



140    //   -> 010 shifted



141    uint8_t count, b;



142    if ((count = readByte(devAddr, regAddr, &b, timeout)) != 0) {



143        uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);



144        b &= mask;



145        b >>= (bitStart - length + 1);



146        *data = b;



147    }



148    return count;



149}

173    }

204 * @param timeout Optional read timeout in milliseconds (0 to disable, leave off to use default class value in I2Cdev::readTimeout)



205 * @return Number of bytes read (-1 indicates failure)



206 */



207int8_t I2Cdev::readBytes(uint8_t devAddr, uint8_t regAddr, uint8_t length, uint8_t *data, uint16_t timeout) {



208    #ifdef I2CDEV_SERIAL_DEBUG



209        Serial.print("I2C (0x");



210        Serial.print(devAddr, HEX);



211        Serial.print(") reading ");

213        Serial.print(" bytes from 0x");

215        Serial.print("...");

218    int8_t count = 0;



219    uint32_t t1 = millis();



220



221    #if (I2CDEV_IMPLEMENTATION == I2CDEV_ARDUINO_WIRE)



222



223        #if (ARDUINO < 100)



224            // Arduino v00xx (before v1.0), Wire library



225



226            // I2C/TWI subsystem uses internal buffer that breaks with large data requests



227            // so if user requests more than BUFFER_LENGTH bytes, we have to do it in



228            // smaller chunks instead of all at once



229            for (uint8_t k = 0; k < length; k += min(length, BUFFER_LENGTH)) {



230                Wire.beginTransmission(devAddr);



231                Wire.send(regAddr);



232                Wire.endTransmission();



233                Wire.beginTransmission(devAddr);



234                Wire.requestFrom(devAddr, (uint8_t)min(length - k, BUFFER_LENGTH));



235



236                for (; Wire.available() && (timeout == 0 || millis() - t1 < timeout); count++) {



237                    data[count] = Wire.receive();



238                    #ifdef I2CDEV_SERIAL_DEBUG



239                        Serial.print(data[count], HEX);



240                        if (count + 1 < length) Serial.print(" ");



241                    #endif



242                }



243



244                Wire.endTransmission();



245            }



246        #elif (ARDUINO == 100)



247            // Arduino v1.0.0, Wire library



248            // Adds standardized write() and read() stream methods instead of send() and receive()



249



250            // I2C/TWI subsystem uses internal buffer that breaks with large data requests



251            // so if user requests more than BUFFER_LENGTH bytes, we have to do it in



252            // smaller chunks instead of all at once



253            for (uint8_t k = 0; k < length; k += min(length, BUFFER_LENGTH)) {



254                Wire.beginTransmission(devAddr);



255                Wire.write(regAddr);



256                Wire.endTransmission();



257                Wire.beginTransmission(devAddr);



258                Wire.requestFrom(devAddr, (uint8_t)min(length - k, BUFFER_LENGTH));



259        



260                for (; Wire.available() && (timeout == 0 || millis() - t1 < timeout); count++) {



261                    data[count] = Wire.read();



262                    #ifdef I2CDEV_SERIAL_DEBUG



263                        Serial.print(data[count], HEX);



264                        if (count + 1 < length) Serial.print(" ");



265                    #endif



266                }



267        



268                Wire.endTransmission();



269            }



270        #elif (ARDUINO > 100)



271            // Arduino v1.0.1+, Wire library



272            // Adds official support for repeated start condition, yay!



273



274            // I2C/TWI subsystem uses internal buffer that breaks with large data requests



275            // so if user requests more than BUFFER_LENGTH bytes, we have to do it in



276            // smaller chunks instead of all at once



277            for (uint8_t k = 0; k < length; k += min(length, BUFFER_LENGTH)) {



278                Wire.beginTransmission(devAddr);



279                Wire.write(regAddr);



280                Wire.endTransmission();



281                Wire.beginTransmission(devAddr);



282                Wire.requestFrom(devAddr, (uint8_t)min(length - k, BUFFER_LENGTH));



283        



284                for (; Wire.available() && (timeout == 0 || millis() - t1 < timeout); count++) {



285                    data[count] = Wire.read();



286                    #ifdef I2CDEV_SERIAL_DEBUG



287                        Serial.print(data[count], HEX);



288                        if (count + 1 < length) Serial.print(" ");



289                    #endif



290                }



291        



292                Wire.endTransmission();



293            }



294        #endif



295



296    #elif (I2CDEV_IMPLEMENTATION == I2CDEV_BUILTIN_FASTWIRE)



297        // Fastwire library (STILL UNDER DEVELOPMENT, NON-FUNCTIONAL!)



298



299        // no loop required for fastwire



300        uint8_t status = Fastwire::readBuf(devAddr, regAddr, data, length);



301        if (status == 0) {



302            count = length; // success



303        } else {



304            count = -1; // error



305        }



306



307    #endif



308



309    // check for timeout



310    if (timeout > 0 && millis() - t1 >= timeout && count < length) count = -1; // timeout



311



312    #ifdef I2CDEV_SERIAL_DEBUG



313        Serial.print(". Done (");



314        Serial.print(count, DEC);



315        Serial.println(" read).");



316    #endif



317



318    return count;



319}

321/** Read multiple words from a 16-bit device register.



322 * @param devAddr I2C slave device address



323 * @param regAddr First register regAddr to read from



324 * @param length Number of words to read



325 * @param data Buffer to store read data in



326 * @param timeout Optional read timeout in milliseconds (0 to disable, leave off to use default class value in I2Cdev::readTimeout)



327 * @return Number of words read (0 indicates failure)



328 */



329int8_t I2Cdev::readWords(uint8_t devAddr, uint8_t regAddr, uint8_t length, uint16_t *data, uint16_t timeout) {



330    #ifdef I2CDEV_SERIAL_DEBUG



331        Serial.print("I2C (0x");



332        Serial.print(devAddr, HEX);



333        Serial.print(") reading ");



334        Serial.print(length, DEC);



335        Serial.print(" words from 0x");



336        Serial.print(regAddr, HEX);



337        Serial.print("...");



338    #endif



339



340    int8_t count = 0;



341    uint32_t t1 = millis();



342



343    #if (I2CDEV_IMPLEMENTATION == I2CDEV_ARDUINO_WIRE)



344



345        #if (ARDUINO < 100)



346            // Arduino v00xx (before v1.0), Wire library



347



348            // I2C/TWI subsystem uses internal buffer that breaks with large data requests



349            // so if user requests more than BUFFER_LENGTH bytes, we have to do it in



350            // smaller chunks instead of all at once



351            for (uint8_t k = 0; k < length * 2; k += min(length * 2, BUFFER_LENGTH)) {



352                Wire.beginTransmission(devAddr);



353                Wire.send(regAddr);



354                Wire.endTransmission();



355                Wire.beginTransmission(devAddr);



356                Wire.requestFrom(devAddr, (uint8_t)(length * 2)); // length=words, this wants bytes



357    



358                bool msb = true; // starts with MSB, then LSB



359                for (; Wire.available() && count < length && (timeout == 0 || millis() - t1 < timeout);) {



360                    if (msb) {



361                        // first byte is bits 15-8 (MSb=15)



362                        data[count] = Wire.receive() << 8;



363                    } else {



364                        // second byte is bits 7-0 (LSb=0)



365                        data[count] |= Wire.receive();



366                        #ifdef I2CDEV_SERIAL_DEBUG



367                            Serial.print(data[count], HEX);



368                            if (count + 1 < length) Serial.print(" ");



369                        #endif



370                        count++;



371                    }



372                    msb = !msb;



373                }



374



375                Wire.endTransmission();



376            }



377        #elif (ARDUINO == 100)



378            // Arduino v1.0.0, Wire library



379            // Adds standardized write() and read() stream methods instead of send() and receive()



380    



381            // I2C/TWI subsystem uses internal buffer that breaks with large data requests



382            // so if user requests more than BUFFER_LENGTH bytes, we have to do it in



383            // smaller chunks instead of all at once



384            for (uint8_t k = 0; k < length * 2; k += min(length * 2, BUFFER_LENGTH)) {



385                Wire.beginTransmission(devAddr);



386                Wire.write(regAddr);



387                Wire.endTransmission();



388                Wire.beginTransmission(devAddr);



389                Wire.requestFrom(devAddr, (uint8_t)(length * 2)); // length=words, this wants bytes



390    



391                bool msb = true; // starts with MSB, then LSB



392                for (; Wire.available() && count < length && (timeout == 0 || millis() - t1 < timeout);) {



393                    if (msb) {



394                        // first byte is bits 15-8 (MSb=15)



395                        data[count] = Wire.read() << 8;



396                    } else {



397                        // second byte is bits 7-0 (LSb=0)



398                        data[count] |= Wire.read();



399                        #ifdef I2CDEV_SERIAL_DEBUG



400                            Serial.print(data[count], HEX);



401                            if (count + 1 < length) Serial.print(" ");



402                        #endif



403                        count++;



404                    }



405                    msb = !msb;



406                }



407        



408                Wire.endTransmission();



409            }



410        #elif (ARDUINO > 100)



411            // Arduino v1.0.1+, Wire library



412            // Adds official support for repeated start condition, yay!



413



414            // I2C/TWI subsystem uses internal buffer that breaks with large data requests



415            // so if user requests more than BUFFER_LENGTH bytes, we have to do it in



416            // smaller chunks instead of all at once



417            for (uint8_t k = 0; k < length * 2; k += min(length * 2, BUFFER_LENGTH)) {



418                Wire.beginTransmission(devAddr);



419                Wire.write(regAddr);



420                Wire.endTransmission();



421                Wire.beginTransmission(devAddr);



422                Wire.requestFrom(devAddr, (uint8_t)(length * 2)); // length=words, this wants bytes



423        



424                bool msb = true; // starts with MSB, then LSB



425                for (; Wire.available() && count < length && (timeout == 0 || millis() - t1 < timeout);) {



426                    if (msb) {



427                        // first byte is bits 15-8 (MSb=15)



428                        data[count] = Wire.read() << 8;



429                    } else {



430                        // second byte is bits 7-0 (LSb=0)



431                        data[count] |= Wire.read();



432                        #ifdef I2CDEV_SERIAL_DEBUG



433                            Serial.print(data[count], HEX);



434                            if (count + 1 < length) Serial.print(" ");



435                        #endif



436                        count++;



437                    }



438                    msb = !msb;



439                }



440        



441                Wire.endTransmission();



442            }



443        #endif



444



445    #elif (I2CDEV_IMPLEMENTATION == I2CDEV_BUILTIN_FASTWIRE)



446        // Fastwire library (STILL UNDER DEVELOPMENT, NON-FUNCTIONAL!)



447



448        // no loop required for fastwire



449        uint16_t intermediate[(uint8_t)length];



450        uint8_t status = Fastwire::readBuf(devAddr, regAddr, (uint8_t *)intermediate, (uint8_t)(length * 2));



451        if (status == 0) {



452            count = length; // success



453            for (uint8_t i = 0; i < length; i++) {



454                data[i] = (intermediate[2*i] << 8) | intermediate[2*i + 1];



455            }



456        } else {



457            count = -1; // error



458        }



459



460    #endif



461



462    if (timeout > 0 && millis() - t1 >= timeout && count < length) count = -1; // timeout



463



464    #ifdef I2CDEV_SERIAL_DEBUG



465        Serial.print(". Done (");



466        Serial.print(count, DEC);



467        Serial.println(" read).");



468    #endif



469    



470    return count;



471}

473/** write a single bit in an 8-bit device register.



474 * @param devAddr I2C slave device address



475 * @param regAddr Register regAddr to write to



476 * @param bitNum Bit position to write (0-7)



477 * @param value New bit value to write



478 * @return Status of operation (true = success)



479 */



480bool I2Cdev::writeBit(uint8_t devAddr, uint8_t regAddr, uint8_t bitNum, uint8_t data) {



481    uint8_t b;



482    readByte(devAddr, regAddr, &b);



483    b = (data != 0) ? (b | (1 << bitNum)) : (b & ~(1 << bitNum));



484    return writeByte(devAddr, regAddr, b);



485}

501/** Write multiple bits in an 8-bit device register.



502 * @param devAddr I2C slave device address



503 * @param regAddr Register regAddr to write to



504 * @param bitStart First bit position to write (0-7)



505 * @param length Number of bits to write (not more than 8)



506 * @param data Right-aligned value to write



507 * @return Status of operation (true = success)



508 */



509bool I2Cdev::writeBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t data) {



510    //      010 value to write



511    // 76543210 bit numbers



512    //    xxx   args: bitStart=4, length=3



513    // 00011100 mask byte



514    // 10101111 original value (sample)



515    // 10100011 original & ~mask



516    // 10101011 masked | value



517    uint8_t b;



518    if (readByte(devAddr, regAddr, &b) != 0) {



519        uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);



520        data <<= (bitStart - length + 1); // shift data into correct position



521        data &= mask; // zero all non-important bits in data



522        b &= ~(mask); // zero all important bits in existing byte



523        b |= data; // combine data with existing byte



524        return writeByte(devAddr, regAddr, b);



525    } else {



526        return false;



527    }



528}

555        return false;

700    void Fastwire::setup(int khz, boolean pullup) {



701        TWCR = 0;



702        #if defined(__AVR_ATmega168__) || defined(__AVR_ATmega8__) || defined(__AVR_ATmega328P__)



703            // activate internal pull-ups for twi (PORTC bits 4 & 5)



704            // as per note from atmega8 manual pg167



705            if (pullup) PORTC |= ((1 << 4) | (1 << 5));



706            else        PORTC &= ~((1 << 4) | (1 << 5));



707        #elif defined(__AVR_ATmega644P__) || defined(__AVR_ATmega644__)



708            // activate internal pull-ups for twi (PORTC bits 0 & 1)



709            if (pullup) PORTC |= ((1 << 0) | (1 << 1));



710            else        PORTC &= ~((1 << 0) | (1 << 1));



711        #else



712            // activate internal pull-ups for twi (PORTD bits 0 & 1)



713            // as per note from atmega128 manual pg204



714            if (pullup) PORTD |= ((1 << 0) | (1 << 1));



715            else        PORTD &= ~((1 << 0) | (1 << 1));



716        #endif



717



718        TWSR = 0; // no prescaler => prescaler = 1



719        TWBR = ((16000L / khz) - 16) / 2; // change the I2C clock rate



720        TWCR = 1 << TWEN; // enable twi module, no interrupt



721    }

723    byte Fastwire::write(byte device, byte address, byte value) {



724        byte twst, retry;



725



726        retry = 2;



727        do {



728            TWCR = (1 << TWINT) | (1 << TWEN) | (1 << TWSTO) | (1 << TWSTA);



729            if (!waitInt()) return 1;



730            twst = TWSR & 0xF8;



731            if (twst != TW_START && twst != TW_REP_START) return 2;



732



733            TWDR = device & 0xFE; // send device address without read bit (1)



734            TWCR = (1 << TWINT) | (1 << TWEN);



735            if (!waitInt()) return 3;



736            twst = TWSR & 0xF8;



737        } while (twst == TW_MT_SLA_NACK && retry-- > 0);



738        if (twst != TW_MT_SLA_ACK) return 4;



739



740        TWDR = address; // send data to the previously addressed device



741        TWCR = (1 << TWINT) | (1 << TWEN);



742        if (!waitInt()) return 5;



743        twst = TWSR & 0xF8;



744        if (twst != TW_MT_DATA_ACK) return 6;



745



746        TWDR = value; // send data to the previously addressed device



747        TWCR = (1 << TWINT) | (1 << TWEN);



748        if (!waitInt()) return 7;



749        twst = TWSR & 0xF8;



750        if (twst != TW_MT_DATA_ACK) return 8;



751



752        return 0;



753    }

755    byte Fastwire::readBuf(byte device, byte address, byte *data, byte num) {



756        byte twst, retry;



757



758        retry = 2;



759        do {



760            TWCR = (1 << TWINT) | (1 << TWEN) | (1 << TWSTO) | (1 << TWSTA);



761            if (!waitInt()) return 16;



762            twst = TWSR & 0xF8;



763            if (twst != TW_START && twst != TW_REP_START) return 17;

765            TWDR = device & 0xfe; // send device address to write



766            TWCR = (1 << TWINT) | (1 << TWEN);



767            if (!waitInt()) return 18;



768            twst = TWSR & 0xF8;



769        } while (twst == TW_MT_SLA_NACK && retry-- > 0);



770        if (twst != TW_MT_SLA_ACK) return 19;



771



772        TWDR = address; // send data to the previously addressed device



773        TWCR = (1 << TWINT) | (1 << TWEN);



774        if (!waitInt()) return 20;



775        twst = TWSR & 0xF8;



776        if (twst != TW_MT_DATA_ACK) return 21;



777



778        /***/



779



780        retry = 2;



781        do {



782            TWCR = (1 << TWINT) | (1 << TWEN) | (1 << TWSTO) | (1 << TWSTA);



783            if (!waitInt()) return 22;

785            if (twst != TW_START && twst != TW_REP_START) return 23;

787            TWDR = device | 0x01; // send device address with the read bit (1)



788            TWCR = (1 << TWINT) | (1 << TWEN);

801            if (twst != TW_MR_DATA_ACK && twst != TW_MR_DATA_NACK) return twst;



802            data[i] = TWDR;



803        }



804



805        return 0;

1076    SIGNAL(TWI_vect) {



1077        switch (TW_STATUS) {



1078            // All Master



1079            case TW_START:     // sent start condition



1080            case TW_REP_START: // sent repeated start condition



1081                // copy device address and r/w bit to output register and ack



1082                TWDR = twi_slarw;



1083                twi_reply(1);



1084                break;



1085    



1086            // Master Transmitter



1087            case TW_MT_SLA_ACK:  // slave receiver acked address



1088            case TW_MT_DATA_ACK: // slave receiver acked data



1089                // if there is data to send, send it, otherwise stop



1090                if (twi_masterBufferIndex < twi_masterBufferLength) {



1091                    // copy data to output register and ack



1092                    TWDR = twi_masterBuffer[twi_masterBufferIndex++];



1093                    twi_reply(1);



1094                } else {



1095                    twi_stop();



1096                }



1097                break;



1098



1099            case TW_MT_SLA_NACK:  // address sent, nack received



1100                twi_error = TW_MT_SLA_NACK;



1101                twi_stop();



1102                break;



1103



1104            case TW_MT_DATA_NACK: // data sent, nack received



1105                twi_error = TW_MT_DATA_NACK;



1106                twi_stop();



1107                break;



1108



1109            case TW_MT_ARB_LOST: // lost bus arbitration



1110                twi_error = TW_MT_ARB_LOST;



1111                twi_releaseBus();



1112                break;



1113    



1114            // Master Receiver



1115            case TW_MR_DATA_ACK: // data received, ack sent



1116                // put byte into buffer



1117                twi_masterBuffer[twi_masterBufferIndex++] = TWDR;



1118



1119            case TW_MR_SLA_ACK:  // address sent, ack received



1120                // ack if more bytes are expected, otherwise nack



1121                if (twi_masterBufferIndex < twi_masterBufferLength) {
